local function runTeleport(targetPos)
    local targetCFrame = CFrame.new(targetPos) + Vector3.new(0, 4, 0)
    local character = Player.Character
    if character and character:FindFirstChild("HumanoidRootPart") then
        local vehicle = nil
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("VehicleSeat") or obj:IsA("Seat") then
                if obj.Occupant == character:FindFirstChildOfClass("Humanoid") then
                    vehicle = obj.Parent
                    break
                end
            end
        end
        
        if vehicle and vehicle:IsA("Model") then
            vehicle:PivotTo(targetCFrame)
            return 
        end
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.Health = 0
            local newCharacter = Player.CharacterAdded:Wait()
            local newRoot = newCharacter:WaitForChild("HumanoidRootPart", 10)
            if newRoot then
                task.wait(0.2)
                newRoot.CFrame = targetCFrame
            end
        end
    end
end

local function createEsp(player)
    if player == Player then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "Esp_Billboard"
    billboard.Size = UDim2.new(0, 200, 0, 60)
    billboard.AlwaysOnTop = true
    billboard.ExtentsOffset = Vector3.new(0, 3, 0)
    billboard.Enabled = isEspActive
    local espLabel = Instance.new("TextLabel")
    espLabel.Size = UDim2.new(1, 0, 1, 0)
    espLabel.BackgroundTransparency = 1
    espLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
    espLabel.TextSize = 13
    espLabel.Font = mcFont
    espLabel.TextStrokeTransparency = 0
    espLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    espLabel.Parent = billboard
    espObjects[player] = {Billboard = billboard, Label = espLabel}
    local connection
    connection = RunService.RenderStepped:Connect(function()
        if espObjects[player] == nil then
            connection:Disconnect()
        else
            if not isEspActive or not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") or not player.Character:FindFirstChild("Humanoid") then
                billboard.Enabled = false
                return
            end
            local char = player.Character
            local hrp = char.HumanoidRootPart
            local hum = char.Humanoid
            local heldTool = char:FindFirstChildOfClass("Tool")
            local toolName = heldTool and heldTool.Name or "None"
            local distance = (Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")) and math.floor((Player.Character.HumanoidRootPart.Position - hrp.Position).Magnitude) or 0
            espLabel.Text = string.format("Name: %s\nHP: %d/%d\nTool: %s\nDist: %dm", player.Name, math.floor(hum.Health), math.floor(hum.MaxHealth), toolName, distance)
            if billboard.Adornee ~= hrp then billboard.Adornee = hrp end
            billboard.Enabled = true
        end
    end)
    billboard.Parent = ScreenGui
end

local function removeEsp(player)

        espObjects[player].Billboard:Destroy()
        espObjects[player] = nil
    end
end

local function optimizeObject(object)
    if not isFpsBoostActive then return end
    if object:IsA("Part") or object:IsA("CornerWedgePart") or object:IsA("MeshPart") or object:IsA("WedgePart") then
        if not originalMaterials[object] then
            originalMaterials[object] = {Material = object.Material, CastShadow = object.CastShadow, Reflectance = object.Reflectance}
        end
        object.Material = Enum.Material.SmoothPlastic
        object.CastShadow = false
        object.Reflectance = 0
    elseif object:IsA("Decal") or object:IsA("Texture") or object:IsA("Clothing") or object:IsA("ShirtGraphic") then
        object:Destroy()
    elseif object:IsA("ParticleEmitter") or object:IsA("Trail") or object:IsA("Smoke") or object:IsA("Fire") or object:IsA("Sparkles") or object:IsA("Beam") then
        object.Enabled = false
    elseif object:IsA("Explosion") then
        object.Visible = false
    elseif object:IsA("Light") then
        object.Enabled = false
    elseif object:IsA("Mesh") or object:IsA("SpecialMesh") then
        object:Destroy()
    end
end

        end
        if Terrain then Terrain.WaterWaveSize = 0 Terrain.WaterWaveSpeed = 0 Terrain.WaterReflectance = 0 Terrain.WaterTransparency = 0 end
        for _, obj in pairs(game.Workspace:GetDescendants()) do optimizeObject(obj) end
        settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
        pcall(function() RunService:Set3DRenderingEnabled(false) end)
    else
        FpsBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        FpsBtn.Text = "SUPER EKSTREM FPS Boost: OFF"
        pcall(function() RunService:Set3DRenderingEnabled(true) end)

        for _, effect in pairs(Lighting:GetChildren()) do
            if effect:IsA("PostEffect") or effect:IsA("BloomEffect") or effect:IsA("BlurEffect") or effect:IsA("ColorCorrectionEffect") or effect:IsA("SunRaysEffect") or effect:IsA("DepthOfFieldEffect") then effect.Enabled = true end

        for obj, original in pairs(originalMaterials) do if obj and obj.Parent then obj.Material = original.Material obj.CastShadow = original.CastShadow obj.Reflectance = original.Reflectance end end
        originalMaterials = {}
    end
end)

local function runTeleport(targetPos)
    local targetCFrame = CFrame.new(targetPos) + Vector3.new(0, 4, 0)
    local character = Player.Character
    if character and character:FindFirstChild("HumanoidRootPart") then
        local vehicle = nil
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("VehicleSeat") or obj:IsA("Seat") then
                if obj.Occupant == character:FindFirstChildOfClass("Humanoid") then
                    vehicle = obj.Parent
                    break
                end
            end
        end
        
        if vehicle and vehicle:IsA("Model") then
            vehicle:PivotTo(targetCFrame)
            return 
        end
        
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.Health = 0
            local newCharacter = Player.CharacterAdded:Wait()
            local newRoot = newCharacter:WaitForChild("HumanoidRootPart", 10)
            if newRoot then
                task.wait(0.2)
                newRoot.CFrame = targetCFrame
            end
        end
    end
end

local function createEsp(player)
    if player == Player then return end
    local billboard = Instance.new("BillboardGui")
    billboard.Name = "Esp_Billboard"
    billboard.Size = UDim2.new(0, 200, 0, 60)
    billboard.AlwaysOnTop = true
    billboard.ExtentsOffset = Vector3.new(0, 3, 0)
    billboard.Enabled = isEspActive
    local espLabel = Instance.new("TextLabel")
    espLabel.Size = UDim2.new(1, 0, 1, 0)
    espLabel.BackgroundTransparency = 1
    espLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
    espLabel.TextSize = 13
    espLabel.Font = mcFont
    espLabel.TextStrokeTransparency = 0
    espLabel.TextStrokeColor3 = Color3.fromRGB(0, 0, 0)
    espLabel.Parent = billboard
    espObjects[player] = {Billboard = billboard, Label = espLabel}
    local connection
    connection = RunService.RenderStepped:Connect(function()
        if espObjects[player] == nil then
            connection:Disconnect()
        else
            if not isEspActive or not player.Character or not player.Character:FindFirstChild("HumanoidRootPart") or not player.Character:FindFirstChild("Humanoid") then
                billboard.Enabled = false
                return
            end
            local char = player.Character
            local hrp = char.HumanoidRootPart
            local hum = char.Humanoid
            local heldTool = char:FindFirstChildOfClass("Tool")
            local toolName = heldTool and heldTool.Name or "None"
            local distance = (Player.Character and Player.Character:FindFirstChild("HumanoidRootPart")) and math.floor((Player.Character.HumanoidRootPart.Position - hrp.Position).Magnitude) or 0
            espLabel.Text = string.format("Name: %s\nHP: %d/%d\nTool: %s\nDist: %dm", player.Name, math.floor(hum.Health), math.floor(hum.MaxHealth), toolName, distance)
            if billboard.Adornee ~= hrp then billboard.Adornee = hrp end
            billboard.Enabled = true
        end
    end)
    billboard.Parent = ScreenGui
end

local function optimizeObject(object)
    if not isFpsBoostActive then return end
    if object:IsA("Part") or object:IsA("CornerWedgePart") or object:IsA("MeshPart") or object:IsA("WedgePart") then
        if not originalMaterials[object] then
            originalMaterials[object] = {Material = object.Material, CastShadow = object.CastShadow, Reflectance = object.Reflectance}
        end
        object.Material = Enum.Material.SmoothPlastic
        object.CastShadow = false
        object.Reflectance = 0
    elseif object:IsA("Decal") or object:IsA("Texture") or object:IsA("Clothing") or object:IsA("ShirtGraphic") then
        object:Destroy()
    elseif object:IsA("ParticleEmitter") or object:IsA("Trail") or object:IsA("Smoke") or object:IsA("Fire") or object:IsA("Sparkles") or object:IsA("Beam") then
        object.Enabled = false
    elseif object:IsA("Explosion") then
        object.Visible = false
    elseif object:IsA("Light") then
        object.Enabled = false
    elseif object:IsA("Mesh") or object:IsA("SpecialMesh") then
        object:Destroy()
    end
end

local function checkInventoryItem(nameENG, nameID)
    if Player.Backpack:FindFirstChild(nameENG) or Player.Backpack:FindFirstChild(nameID) then return true
    elseif Player.Character and (Player.Character:FindFirstChild(nameENG) or Player.Character:FindFirstChild(nameID)) then return true end
    return false
end

local function equipItem(nameENG, nameID)
    local target = Player.Backpack:FindFirstChild(nameENG) or Player.Backpack:FindFirstChild(nameID)
    if target then
        local humanoid = Player.Character and Player.Character:FindFirstChildOfClass("Humanoid")
        if humanoid then humanoid:EquipTool(target) return true end
    elseif Player.Character:FindFirstChild(nameENG) or Player.Character:FindFirstChild(nameID) then return true end
    return false
end

local function countItem(nameENG, nameID)
    if not Player then return 0 end
    local count = 0
    for _, item in pairs(Player.Backpack:GetChildren()) do
        if (item.Name:lower() == nameENG:lower() or item.Name:lower() == nameID:lower()) and item:IsA("Tool") then count = count + 1 end
    end
    if Player.Character then
        local heldItem = Player.Character:FindFirstChildOfClass("Tool")
        if heldItem and (heldItem.Name:lower() == nameENG:lower() or heldItem.Name:lower() == nameID:lower()) then count = count + 1 end
    end
    return count
end
